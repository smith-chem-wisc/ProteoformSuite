80  double apply(SpectrumType & spectrum)
81  {
82  if (spectrum.size() < 2)
83  {
84  return 0;
85  }
86  double tolerance = (double)param_.getValue("tolerance");
87  double parentmass = 0.0;
88  if (!spectrum.getPrecursors().empty()) parentmass = spectrum.getPrecursors()[0].getMZ();
89  double result(0);
90 
91  spectrum.sortByPosition();
92 
94  Size j = spectrum.size() - 1;
95  for (Size i = 0; i < spectrum.size() && i <= j; /*++i*/)
96  {
97  double sum = spectrum[i].getPosition()[0] + spectrum[j].getPosition()[0];
98 
99  if (std::fabs(sum - parentmass) < tolerance)
100  {
101  result += spectrum[i].getIntensity() + spectrum[j].getIntensity();
102  }
103 
104  if (sum < parentmass)
105  {
106  ++i;
107  }
108  else
109  {
110  if (sum > parentmass)
111  {
112  --j;
113  }
114  }
115  }
116 
117  return result;
118  }
