82  {
83  std::map<double, bool> marked;
84  (*cvit)->apply(marked, spectrum);
85  for (std::map<double, bool>::const_iterator cmit = marked.begin(); cmit != marked.end(); ++cmit)
86  {
87  if (cmit->second)
88  {
89  marks[cmit->first]++;
90  }
91  }
92  }
93 
94  for (Iterator it = spectrum.begin(); it != spectrum.end(); )
95  {

99  }
