116  inline int OPENMS_DLLAPI lowess(const std::vector<double>& x, const std::vector<double>& y,
117  std::vector<double>& result)
118  {
119  OPENMS_PRECONDITION(x.size() == y.size(), "Vectors x and y must have the same length")
120  OPENMS_PRECONDITION(x.size() >= 2, "Need at least two points for smoothing")
121  OPENMS_PRECONDITION(std::adjacent_find(x.begin(), x.end(), std::greater<double>()) == x.end(),
122  "The vector x needs to be sorted")
123 
124  double delta = 0.01 * (x[ x.size()-1 ] - x[0]); // x is sorted
125  return lowess(x, y, 2.0/3, 3, delta, result);
