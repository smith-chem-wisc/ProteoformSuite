193  template <typename T>
194  T gcd(T a, T b, T & u1, T & u2)
195  {
196  u1 = 1;
197  u2 = 0;
198  T u3 = a;
199 
200  T v1 = 0;
201  T v2 = 1;
202  T v3 = b;
203 
204  while (v3 != 0)
205  {
206  T q = u3 / v3;
207  T t1 = u1 - v1 * q;
208  T t2 = u2 - v2 * q;
209  T t3 = u3 - v3 * q;
210 
211  u1 = v1;
212  u2 = v2;
213  u3 = v3;
214 
215  v1 = t1;
216  v2 = t2;
217  v3 = t3;
218  }
219 
220  return u3;
221  }

296  inline static std::pair<double, double> getTolWindow(double val, double tol, bool ppm)
297  {
298  double left, right;
299 
300  if (ppm)
301  {
302  left = val - val * tol * 1e-6;
303  right = val / (1.0 - tol * 1e-6);
304  }
