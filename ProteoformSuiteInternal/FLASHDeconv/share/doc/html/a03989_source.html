142  {
143  type.clear();

148  if (in_type == FileTypes::UNKNOWN)

151  }
152  type = FileTypes::typeToName(in_type);
153  fh.loadExperiment(in_filename, exp, in_type, ProgressLogger::NONE, false, false);
