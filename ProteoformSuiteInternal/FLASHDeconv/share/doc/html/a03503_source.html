59  template <class PtrType>
60  inline bool cmpPtrSafe(const PtrType& a, const PtrType& b)
61  {
62  // We are not interested whether the pointers are equal but whether the
63  // contents are equal
64  if (a == nullptr && b == nullptr)
65  {
66  return true;
67  }
68  else if (a == nullptr || b == nullptr)
69  {
70  return false; // one is null the other is not
71  }
72  else
73  {
74  // compare the internal object
75  return (*a == *b);
76  }
77  }

82  template <class ContainerType>
83  inline bool cmpPtrContainer(const ContainerType& a, const ContainerType& b)
84  {
85  if (a.size() != b.size()) return false;
86 
87  // check that all elements of a and b are equal using safe comparison
88  // (taking NULL into account)
89  for (typename ContainerType::size_type i = 0; i < a.size(); i++)
90  {
91  if (!cmpPtrSafe(a[i], b[i]))
92  {
93  return false;
94  }
95  }
96  return true;
97  }
98 
99  }
