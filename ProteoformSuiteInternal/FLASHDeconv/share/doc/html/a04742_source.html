83  QualityType fitOffset_(InterpolationModel * model, const RawDataArrayType & set, const CoordinateType stdev1, const CoordinateType stdev2, const CoordinateType offset_step)
84  {
85  const CoordinateType offset_min = model->getInterpolation().supportMin() - stdev1;
86  const CoordinateType offset_max = model->getInterpolation().supportMin() + stdev2;
87 
88  CoordinateType offset;
89  QualityType correlation;
90 
91  //test model with default offset
92  std::vector<float> real_data;
93  real_data.reserve(set.size());
94  std::vector<float> model_data;
95  model_data.reserve(set.size());
96 
97  for (Size i = 0; i < set.size(); ++i)
98  {
99  real_data.push_back(set[i].getIntensity());
100  model_data.push_back(model->getIntensity(DPosition<1>(set[i].getPosition())));
101  }
102 
103  CoordinateType max_offset = model->getInterpolation().getOffset();
104  QualityType max_correlation = Math::pearsonCorrelationCoefficient(real_data.begin(), real_data.end(), model_data.begin(), model_data.end());
105 
106  //test different offsets
107  for (offset = offset_min; offset <= offset_max; offset += offset_step)
108  {
109  // set offset
110  model->setOffset(offset);
111 
112  // get samples
113  model_data.clear();
114  for (Size i = 0; i < set.size(); ++i)
115  {
116  model_data.push_back(model->getIntensity(DPosition<1>(set[i].getPosition())));
117  }
118 
119  correlation = Math::pearsonCorrelationCoefficient(real_data.begin(), real_data.end(), model_data.begin(), model_data.end());
120 
121  if (correlation > max_correlation)
122  {
123  max_correlation = correlation;
124  max_offset = offset;
125  }
126  }
127 
128  model->setOffset(max_offset);
129 
130  return max_correlation;
131  }
