298  template <typename MAPTYPE> String resolveAutoMode_(const MAPTYPE& map) const
299  {
300  String ion_mode_internal;
301  String ion_mode_detect_msg = "";
302  if (map.size() > 0)
303  {
304  if (map[0].metaValueExists("scan_polarity"))
305  {
306  StringList pols = ListUtils::create<String>(String(map[0].getMetaValue("scan_polarity")), ';');
307  if (pols.size() == 1 && pols[0].size() > 0)
308  {
309  pols[0].toLower();
310  if (pols[0] == "positive" || pols[0] == "negative")
311  {
312  ion_mode_internal = pols[0];
313  OPENMS_LOG_INFO << "Setting auto ion-mode to '" << ion_mode_internal << "' for file " << File::basename(map.getLoadedFilePath()) << std::endl;
314  }
315  else ion_mode_detect_msg = String("Meta value 'scan_polarity' does not contain unknown ion mode") + String(map[0].getMetaValue("scan_polarity"));
316  }
317  else
318  {
319  ion_mode_detect_msg = String("ambiguous ion mode: ") + String(map[0].getMetaValue("scan_polarity"));
320  }
321  }
322  else
323  {
324  ion_mode_detect_msg = String("Meta value 'scan_polarity' not found in (Consensus-)Feature map");
325  }
326  }
327  else
328  { // do nothing, since map is
329  OPENMS_LOG_INFO << "Meta value 'scan_polarity' cannot be determined since (Consensus-)Feature map is empty!" << std::endl;
330  }
331 
332  if (ion_mode_detect_msg.size() > 0)
333  {
334  throw Exception::InvalidParameter(__FILE__, __LINE__, OPENMS_PRETTY_FUNCTION, String("Auto ionization mode could not resolve ion mode of data (") + ion_mode_detect_msg + "!");
335  }
336 
337  return ion_mode_internal;
338  }
